  def self.ensureConfig( session, logger )
    return if @@config

    @@config = session[:multiSchedConfig] # ELABORATE for >1 grid/session XXXX
    
    if @@config
      # logger.info "\nConfig from session.\n#{@@config.inspect}"
      return
    end

    SchedResource.configFromYaml( session, logger )
  end
