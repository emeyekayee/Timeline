def Timelabel.get_timeblocks(id, t1, t2, inc)
  klass = self
  it0, it2, itb = [klass.floor(t1), t2, klass.t_block].map(&:to_i)

  it0 += itb if inc == 'hi'
  it2 -= itb if inc == 'lo'

  it0.step(it2, itb).map{|i| klass.new(Time.at i)}
end
