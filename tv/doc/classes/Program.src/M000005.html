  def set_visual_info ()
    classes = ''

    ct = self.category_type
    ct  &&  ct !~ /unknown/i  &&
      classes << " type_" + ct.gsub( /[^a-zA-Z0-9\-_]+/, '_' )

    classes << " " + to_css_class(self.category)

    self.css_classes = classes
    self.block_label = "#{self.title}:<br/>&nbsp;#{self.subtitle}"
  end
