  def Timelabel.get_timeblocks(id, t1, t2)
    v = Timeheader.variantOfId id
    # v = 'dayNight'
    tb = t_blockForVariant v
    
    res = []

    t = t1
    if tb > 1.hour # Generalize this through :day, :month, :year  XXXX
      t = t1.change( :hour => 0 )
      until (t..(t + tb)) === t; t += tb end
    end

    while ( t < t2 ) do
      res.push( Timelabel.new(v, t) )
      t += tb
    end

    res
  end
