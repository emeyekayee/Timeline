def schedule
  SchedResource.send( params[:reset] ? :config_from_yaml : :ensure_config,
                      session )

  @t1 = params[:t1] || time_default
  @t2 = params[:t2] || @t1 + SchedResource.visible_time
  @inc= params[:inc]
  get_data_for_time_span
end
